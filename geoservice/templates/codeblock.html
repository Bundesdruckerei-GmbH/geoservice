            	import { MapContainer, Marker } from 'react-leaflet'
            	import { GeoserviceTiles, GeoJSON } from "@dataapplication/react-geoservice"
            
            	import "leaflet/dist/leaflet.css"
            
            	function App() {
            
            			const [ result ] = usePromise(
            				() => fetch('stats.json').then(r => r.json()),
            			[]
            		)
            
            		function getColor(code: string): string {
            			if (!result || !result[code]) {
            				return 'gray'
            			}
            			const value = 255 * (1 - (result[code] / 80000000));
            			return `rgb(${value}, 255, ${value})`
            		}
            
            		return (
            			<Mapcontainer
            				className="map"
            				zoom={ 2 }
            				minZoom={ 2 }
            				maxZoom={ 13 }
            				center={ { lat: 19.5, lng: 27 } }
            			>
            			<Marker
            					title="Burgermeister"
            					position={ {
            						lat: 52.50111,
            						lng: 13.44230
            					} }
            			/>
            			<GeoserviceTiles
            					feature={ country =>
            						!!(result ?? {})[country.properties.ADMIN_GID_0]] &&
            						<GeoJSON
            							data={ country }
            							style={ {
            								weight: 1,
            								fillOpacity: 1,
            								fillColor: getColor(country.properties.ADM_GID_0),
            							} }
            						/>
            					}
            
            				/>
            		</Mapcontainer>
            	)
            }
            
            export default App;
{% endmacro %}
